## 函数作为参数

> 将核心逻辑传入方法体，体现开闭原则

```python
"""
    函数式编程思想
"""

class SkillData:
    def __init__(self,id,name,atk_ratio,duration):
        self.id = id
        self.name = name
        self.atk_ratio = atk_ratio
        self.duration = duration

    def __str__(self):
        return "技能:%d,%s,%d,%d"%(self.id,self.name,self.atk_ratio,self.duration)

list_skill = [
    SkillData(101,'乾坤大挪移',5,10),
    SkillData(102,'降龙十八掌',8,5),
    SkillData(103,'葵花宝典',10,2)
]

# 需求1：获取攻击比例大于6的所有技能
def find01():
    for item in list_skill:
        if item.atk_ratio > 6:
            yield item

# 需求2：获取持续时间在4--11之间所有技能
def find02():
    for item in list_skill:
        if 4<item.duration<11:
            yield item            

# 需求3：获取技能名称大于4个字并且持续时间小于6的所有技能  
def find04():
    for item in list_skill:
        if len(item.name) > 4 and item.duration < 6:
            yield item

#将每个变化的条件，单独定义在函数中
def condition1(item):
    return item.atk_ratio > 6

def condition2(item):
    return 4<item.duration<11 

def condition1(item):
    return len(item.name) > 4 and item.duration < 6

# "继承"（隔离变化）
def find(func_condition):
    for item in list_skill:
        #"多态"：调用父(变量),执行子(具体函数)
        #        不同子类重写父类方法，执行逻辑不同
        if func_condition(item):
            yield item  

for item in find(condition2):
    print(item)
```

### 闭包

1. 三要素:

   a.必须有一个内嵌函数

   b.内嵌函数必须引用外部 函数中变量

   c.外部函数返回值必须是内嵌函数

2. > 语法：def fun01():
   >
   >   a = 1
   >
   >   def fun02():
   >
   > ​    print(a)
   >
   > 
   >
   >   return fun02
   >
   > 
   >
   > \# 调用外部函数,返回值是内嵌函数
   >
   > result = fun01()
   >
   > \# 调用内嵌函数
   >
   > result()

3. > 定义：在一个函数内部的函数，同时内部函数又引用了外部函数的变量

4. > 本质：闭包是将内部函数和外部函数的执行环境绑定在一起的对象

```python
ef fun01():
    a = 1
    def fun02():
        print(a)

    return fun02

# 调用外部函数,返回值是内嵌函数
result = fun01()
# 调用内嵌函数
result() #可以访问外部变量a

# 闭包应用:逻辑连续，当内部函数被调用时，不脱离当前逻辑
# 压岁钱
def give_gife_money(money):
    print("得到了%d"%money)
    def child_buy(target,price):
        nonlocal money
        if money >= price:
            money -= price
            print('花了%.1f，买了%s'%(price,target))
        else:
            print('钱不够')

    return child_buy

# 下列代码是一个连续的逻辑
action = give_gife_money(10000)
action('老鼠肉',0.5)
action('车',2000)
action('手机',8000) 
```

```python
    def wrapper(*args,**kwags):
        print('权限验证')
        func(*args,**kwags)

    return wrapper

def enter_background(login_id,pwd):
```
